Change all repos to Denmark. We need to a build script - OpenEuler 25.03 - KDE

sudo nano byt-mirror.sh

Copy and paste This script to terminal.

----------------------------------------

#!/bin/bash
# Byt mirrors till dotsrc för openEuler 25.03 x86_64

REPO_DIR="/etc/yum.repos.d"
NEW_MIRROR="https://mirrors.dotsrc.org/openeuler/openEuler-25.03"

echo "Byter mirrors i $REPO_DIR till $NEW_MIRROR för x86_64..."

for repo in "$REPO_DIR"/*.repo; do
    echo "Uppdaterar $repo..."
    sed -i.bak -E \
        "s|https://mirrors\.openeuler\.org/openeuler[^/]*/x86_64/|$NEW_MIRROR/x86_64/|g" \
        "$repo"
done

echo "Klart! Backup-filer skapade med .bak-tillägg."

---------------------------------------

Run this in terminal: sudo chmod +x byt-mirror.sh

Run Script: sudo ./byt-mirror.sh
