Make this script for KDE 6 to start for SPARKYLINUX.

Copy and paste to terminal: sudo nano ~/start-kde6.sh

NOW COPY AND PASTE THIS TO TERMINAL =) TAKE ALL - SPARKYLINUX START SCRIPT FOR KDE 6

#!/usr/bin/env bash

# ----- X410 / X11 -----
export DISPLAY=${DISPLAY:-$(awk '/nameserver/ {print $2; exit}' /etc/resolv.conf):0.0}
export QT_QPA_PLATFORM=xcb
export GDK_BACKEND=x11
export SDL_VIDEODRIVER=x11
export QT_X11_NO_MITSHM=1
unset WAYLAND_DISPLAY

# ----- Runtime dir för KDE -----
export XDG_RUNTIME_DIR=${XDG_RUNTIME_DIR:-/tmp/runtime-$USER}
mkdir -p "$XDG_RUNTIME_DIR"
chmod 700 "$XDG_RUNTIME_DIR"

# ----- Ljud via WSLg PulseAudio -----
for s in \
  /mnt/wslg/PulseServer \
  /mnt/wslg/runtime-dir/pulse/native \
  "/mnt/wslg/run/user/$(id -u)/pulse/native"
do
  if [ -S "$s" ]; then
    export PULSE_SERVER="unix:$s"
    break
  fi
done

echo "DISPLAY=$DISPLAY"
echo "PULSE_SERVER=$PULSE_SERVER"

pactl info | sed -n '1,10p' || echo "PulseAudio hittades inte ännu."

exec dbus-run-session startplasma-x11
EOF
